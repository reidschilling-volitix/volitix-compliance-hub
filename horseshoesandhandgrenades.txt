const NotamTab = ({ notams, isEditing, state, setState, derived, generate, onSubmit, onCancel, onDelete, notify, company, certs }) => {
  const kmlRef = useRef(null);
  const handleKml = (e) => {
    const file = e.target.files[0]; if (!file) return;
    const r = new FileReader(); r.onload = (ev) => setState(p => ({...p, kmlData: ev.target.result, kmlFileName: file.name})); r.readAsText(file);
  };
  if (isEditing) {
    return (
      <FormCard title={state.id ? 'Edit NOTAM' : 'Create NOTAM'} icon={Navigation} onSubmit={onSubmit} onCancel={onCancel} submitLabel="Archive NOTAM">
        <div className="grid grid-cols-1 md:grid-cols-2 gap-6 min-w-0">
          <Input type="date" label="NOTAM Date" value={state.date} onChange={e => setState({...state, date: e.target.value})} required/>
          <Input label="NOTAM Number" placeholder="e.g. FDC 1/2345" value={state.notamNumber} onChange={e => setState({...state, notamNumber: e.target.value})}/>
          <Input type="time" label="Local Start Time" value={state.localStartTime} onChange={e => setState({...state, localStartTime: e.target.value})}/>
          <Input label="Duration (Hours)" type="number" step="any" value={state.duration} onChange={e => setState({...state, duration: e.target.value})}/>
          <Input label="Radius (NM)" type="number" step="any" value={state.radius} onChange={e => setState({...state, radius: e.target.value})} required/>
        </div>
        <GeoAndKmlInput notify={notify} state={state} setState={setState} kmlRef={kmlRef} handleKml={handleKml} />
        <div className="bg-slate-950 p-6 rounded-2xl border border-[#9cd33b]/30 space-y-3">
          <p className="text-[10px] text-[#9cd33b] font-black uppercase tracking-widest">Generated Script</p>
          <div className="bg-slate-900 p-4 rounded-xl border border-slate-800 font-mono text-[9px] text-slate-400 break-words whitespace-pre-wrap max-h-32 overflow-y-auto">{generate()}</div>
          <Button variant="secondary" onClick={() => copyToClipboard(generate())} className="w-full"><Paperclip size={14}/> Copy Script</Button>
        </div>
      </FormCard>
    );
  }
  return (
    <TableCard title="NOTAM Archive" icon={Navigation} actionLabel="NOTAM" onAction={() => setState({...defaultNotamState, isEditing: true})}>
      <table className="w-full text-left min-w-[700px]">
        <thead className={tw.thead}>
          <tr><th className={tw.th}>Date / Number</th><th className={tw.th}>Location / Radius</th><th className={tw.th}>Duration</th><th className={`${tw.th} text-right`}>Actions</th></tr>
        </thead>
        <tbody className="divide-y divide-slate-800/40">
          {notams.length === 0 && <tr><td colSpan="4" className="p-8 text-center text-slate-500 font-bold uppercase tracking-widest text-[10px]">No NOTAMs archived.</td></tr>}
          {notams.map(n => (
            <tr key={n.id} className={tw.tr}>
              <td className={tw.td}><p className="font-mono text-[#9cd33b] text-xs font-black break-words">{String(n.date)}</p><p className="text-[10px] text-slate-400 uppercase mt-1 tracking-widest break-words">{String(n.notamNumber || 'N/A')}</p></td>

              <ActionButtons onEdit={() => setState({...n, isEditing: true})} onDelete={() => onDelete('notam_logs', n.id)} />
            </tr>
          ))}
        </tbody>
      </table>
    </TableCard>
  );
};

const FaaReportTab = ({ company, fleet, certs, monthlyReport, reportMonth, setReportMonth, showFormalReport, setShowFormalReport, notify, faaReports, onDelete }) => {
  const handleGenerateFormalReport = async () => {
    if (!monthlyReport.logs || monthlyReport.logs.length === 0) return notify("No flights logged for this month.", "error");
    const report = {
      month: reportMonth,
      companyName: company.name,
      picName: company.supervisor,
      partNumber: company.certNo || 'N/A',
      flightCount: monthlyReport.flightCount,
      totalHours: monthlyReport.totalHours,
      logs: monthlyReport.logs
    };
    try {
      const db_ref = collection(db, 'artifacts', appId, 'public', 'data', 'faa_reports');
      await addDoc(db_ref, report);
      notify("Report filed with database.", "success");
      setShowFormalReport(false);
    } catch(err) {
      notify("Error saving report.", "error");
    }
  };

  const handlePrintReport = () => {
    const reportContent = document.getElementById('faa-report-content');
    if (!reportContent) return notify("Report not ready.", "error");
    const printWindow = window.open('', '', 'width=900,height=1200');
    printWindow.document.write(reportContent.innerHTML);
    printWindow.document.close();
    printWindow.print();
  };

  return (
    <div className="space-y-8 animate-fade-in min-w-0">
      <Card className="space-y-6">
        <div className="flex flex-col md:flex-row md:justify-between md:items-center gap-4 border-b border-slate-800 pb-6">
          <div className="flex items-center gap-4">
            <FileBarChart size={32} className="text-[#9cd33b]"/>
            <div className="min-w-0">
              <h2 className="text-2xl font-black uppercase tracking-tight text-white truncate">FAA Part 137 Monthly Report</h2>
              <p className="text-xs text-slate-400 font-bold uppercase tracking-widest mt-1 truncate">Compliance & Flight Hours</p>
            </div>
          </div>
          <Input type="month" label="Report Month" value={reportMonth} onChange={e => setReportMonth(e.target.value)} className="w-full md:w-40"/>

        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
          <Card className="p-4 flex flex-col gap-2"><p className={tw.label}>Total Flights</p><p className="text-4xl font-black text-[#9cd33b]">{monthlyReport.flightCount}</p></Card>
          <Card className="p-4 flex flex-col gap-2"><p className={tw.label}>Total Hours</p><p className="text-4xl font-black text-blue-400">{monthlyReport.totalHours}</p></Card>

        </div>

        <div className="flex flex-col md:flex-row gap-4 md:items-center md:justify-between">
          <div className="space-y-1">
            <p className="text-[10px] text-slate-400 font-black uppercase tracking-widest">Organization: {String(company.name)}</p>

            <p className="text-[10px] text-slate-400 font-black uppercase tracking-widest">Part 137 Cert: {String(company.certNo || 'N/A')}</p>
          </div>
          <div className="flex gap-3 flex-wrap">
            <Button onClick={() => setShowFormalReport(true)} disabled={monthlyReport.flightCount === 0}><FileText size={14}/> Review Report</Button>
            <Button variant="secondary" onClick={handlePrintReport} disabled={monthlyReport.flightCount === 0}><Printer size={14}/> Print</Button>
          </div>

      </Card>
